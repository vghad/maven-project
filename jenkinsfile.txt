pipeline {
    agent any
    tools { 
        maven 'localmaven' 
        jdk 'localJDK' 
    }
    stages {
        stage ('Initialize') {
            steps {
                sh '''
                    echo "PATH = ${PATH}"
                    echo "M2_HOME = ${M2_HOME}"
                ''' 
            }
        }

        stage ('Build') {
            steps {
            git url: 'https://github.com/cyrille-leclerc/multi-module-maven-project'
            sh 'mvn clean package'
            echo 'Build job'
            }
        }
    }
}
